# Steps

## Install

Integrate with your build to [auto-import](/#integrating), or add an import in your component:

```
import {ASteps, AStep, AStepTitle, AStepDescription, AButton} from "@cisco-sbg-ui/atomic-react";
```

## Horizontal Steps

<Playground
  code={`() => {
    const [active, setActive] = useState(1);
    const steps = 3;
    return (
      <div>
        <div style={{border: "solid 2px #333"}} className="my-2 pa-2">
          {active > steps ? "Completed" : "Step " + active}
        </div>
        <h3>Steps</h3>
        <ASteps>
          <AStep stepNumber={1} active={active === 1} visited={active > 1}>
            <AStepTitle>Title </AStepTitle>
            <AStepDescription>
              Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit.
            </AStepDescription>
          </AStep>
          <AStep
            stepNumber={2}
            active={active === 2}
            visited={active > 2}
            disabled={true}>
            <AStepTitle>Title</AStepTitle>
            <AStepDescription>
              Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit.
            </AStepDescription>
          </AStep>
          <AStep stepNumber={3} active={active === 3} visited={active > 3}>
            <AStepTitle>Title</AStepTitle>
            <AStepDescription>
                Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit.
            </AStepDescription>
          </AStep>
        </ASteps>
        <AButton onClick={() => active > 1 && setActive(active - 1)}>
          Previous
        </AButton>
        <AButton onClick={() => active <= steps && setActive(active + 1)}>
          Next
        </AButton>
      </div>
    );
  }`}
/>

## Steps used with json data

<Playground
  code={`() => {
    const [active, setActive] = useState(1);
    const data = [
      {
        title: "Active. Long Title.",
        description:
          "Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
      },
      {
        title: "Title",
        description:
          "Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit."
      },
      {
        title: "Default",
        description:
          "Description of title. Lorem ipsum dolor sit amet, consectetur adipiscing elit."
      }
    ];
    return (
      <div>
        <div style={{border: "solid 2px #333"}} className="my-2 pa-2">
          {active > data.length ? "Completed" : "Step " + active}
        </div>
        <h3>Steps with icon:</h3>
        <ASteps>
          {data.map((item, index) => {
            const stepNumber = index + 1;
            return (
              <AStep
                key={stepNumber}
                stepNumber={stepNumber}
                active={active === stepNumber}
                visited={active > stepNumber}
                disabled={stepNumber === 2}
                showIconOnSisited={true}>
               <AStepTitle>
                  {item.title}
                </AStepTitle>
                <AStepDescription>{item.description}</AStepDescription>
               </AStep>
            );
          })}
        </ASteps>
        <AButton onClick={() => active > 1 && setActive(active - 1)}>
          Previous
        </AButton>
        <AButton onClick={() => active <= data.length && setActive(active + 1)}>
          Next
        </AButton>
      </div>
    );
  }`}
/>
