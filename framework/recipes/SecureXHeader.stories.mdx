# Recipe: SecureX Header

## Install

The component is available on [Cisco's Artifactory package registry](https://devhub.cisco.com/artifactory/webapp/#/home).

```
  npm i --save @cisco-ats/atomic-react
```

And in the target component add:

```
import AApp from "@cisco-ats/atomic-react/AApp";
import {ADropdown, ADropdownMenu, ADropdownItem} from "@cisco-ats/atomic-react/ADropdown";
import {AHeader, AHeaderLogo, AHeaderTitle, AHeaderNavigation, AHeaderToolbar, AHeaderToolbarItem} from "@cisco-ats/atomic-react/AHeader";
import AIcon from "@cisco-ats/atomic-react/AIcon";
import {ATabGroup, ATab} from "@cisco-ats/atomic-react/ATabs";
```
